üîê Row Level Security (RLS) Setup
Essential for multi-tenant data isolation between corporate accounts
-- Enable RLS on all tables
ALTER TABLE profiles ENABLE ROW LEVEL SECURITY;
ALTER TABLE organizations ENABLE ROW LEVEL SECURITY;
ALTER TABLE courses ENABLE ROW LEVEL SECURITY;
ALTER TABLE enrollments ENABLE ROW LEVEL SECURITY;

-- Users can only see their own profile
CREATE POLICY "Users see own profile" ON profiles
  FOR SELECT USING (auth.uid() = id);

-- Corporate admins can see their org's users
CREATE POLICY "Admins see org users" ON profiles
  FOR SELECT USING (
    organization_id IN (
      SELECT organization_id FROM profiles 
      WHERE id = auth.uid() AND role = 'admin'
    )
  );

-- Course access based on enrollment
CREATE POLICY "Users see enrolled courses" ON courses
  FOR SELECT USING (
    id IN (
      SELECT course_id FROM enrollments 
      WHERE user_id = auth.uid()
    )
  );
üõ°Ô∏è Advanced RLS for Corporate Isolation
Prevent data leaks between different organizations
-- Function to get user's organization
CREATE OR REPLACE FUNCTION get_user_org()
RETURNS UUID AS $$
  SELECT organization_id FROM profiles 
  WHERE id = auth.uid()
$$ LANGUAGE SQL STABLE;

-- Org-specific course access
CREATE POLICY "Org courses only" ON courses
  FOR ALL USING (
    organization_id = get_user_org() 
    OR organization_id IS NULL -- Public courses
  );

üîí API Security & Rate Limiting
Protect against abuse and unauthorized access
-- Database function for rate limiting
CREATE OR REPLACE FUNCTION check_rate_limit(
  user_id UUID,
  action_type TEXT,
  max_requests INTEGER DEFAULT 100,
  window_minutes INTEGER DEFAULT 60
) RETURNS BOOLEAN AS $$
DECLARE
  request_count INTEGER;
BEGIN
  SELECT COUNT(*)
  INTO request_count
  FROM api_requests
  WHERE user_id = $1
    AND action_type = $2
    AND created_at > NOW() - INTERVAL '1 minute' * window_minutes;
  
  RETURN request_count < max_requests;
END;
$$ LANGUAGE plpgsql;

// middleware.ts - Global security middleware
import { NextResponse } from 'next/server'
import type { NextRequest } from 'next/server'

export function middleware(request: NextRequest) {
  // Security headers
  const response = NextResponse.next()
  
  // Content Security Policy
  response.headers.set(
    'Content-Security-Policy',
    "default-src 'self'; script-src 'self' 'unsafe-inline' *.supabase.co; style-src 'self' 'unsafe-inline'; img-src 'self' data: *.supabase.co;"
  )
  
  // Security headers
  response.headers.set('X-Frame-Options', 'DENY')
  response.headers.set('X-Content-Type-Options', 'nosniff')
  response.headers.set('Referrer-Policy', 'strict-origin-when-cross-origin')
  
  // HSTS
  response.headers.set(
    'Strict-Transport-Security',
    'max-age=63072000; includeSubDomains; preload'
  )
  
  return response
}

// lib/validation.ts
import { z } from 'zod'
import DOMPurify from 'isomorphic-dompurify'

// Input schemas
export const CourseSchema = z.object({
  title: z.string().min(1).max(200).trim(),
  description: z.string().max(2000).optional(),
  organizationId: z.string().uuid().optional()
})

// Sanitize HTML content
export const sanitizeHtml = (html: string) => {
  return DOMPurify.sanitize(html, {
    ALLOWED_TAGS: ['b', 'i', 'em', 'strong', 'p', 'br', 'ul', 'ol', 'li'],
    ALLOWED_ATTR: []
  })
}

// API route with validation
export default async function handler(req: NextRequest) {
  try {
    const validatedData = CourseSchema.parse(req.body)
    
    // Sanitize text fields
    validatedData.description = sanitizeHtml(validatedData.description)
    
    // Process request...
  } catch (error) {
    if (error instanceof z.ZodError) {
      return NextResponse.json(
        { error: 'Invalid input', details: error.errors },
        { status: 400 }
      )
    }
  }
}

‚ö†Ô∏è CRITICAL SECURITY REMINDERS
1. Service Role Key: NEVER expose Supabase service role key in client-side code. Only use in API routes.

2. RLS Testing: Always test RLS policies with different user accounts before going live.

3. HTTPS Only: Never allow HTTP traffic in production. All sensitive data must be encrypted in transit.

4. Regular Updates: Keep all dependencies updated and monitor security advisories.

5. Backup Strategy: Implement automated backups with encryption for all user data.
